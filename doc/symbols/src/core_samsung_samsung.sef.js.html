 34 Samsung.SEF.loadTV = function(){
 35 	System.Log.verbose("[core/samsung/samsung.sef.js] Samsung.SEF.loadTV()");	
 36 	
 37 	if (Samsung.SEF.TV === null) Samsung.SEF.TV = System.Cache.getHtml("pluginObjectTV");
 38 	
 39 	if (!Samsung.SEF.TV) System.Log.warning("[core/samsung/samsung.sef.js] Cannot Load Samsung SEF TV Plugin");
 40 	else System.Log.info("[core/samsung/samsung.sef.js] Samsung SEF TV Loaded");
 41 };
