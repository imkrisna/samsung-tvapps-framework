 11 /**
 12  * Runtime delay (in miliseconds) from loading to executing framework, Required to provide time for resources
 13  * initializing themselves.
 14  * 
 15  * @type Number
 16  * @default 1000
 17  *

 21 Main.PREPARE_DELAY	= 1000;

 23 /**
 24  * Javascript Timeout object to execute framework construction.
 25  * @type Object
 26  * */
 27 Main.Runner 		= null;
 28 
 29 /**
 30  * Callback function which is called after the index.html loaded
 31  * @description Preparing all required resources and executing the {@link Main.Runner} and load framework.
 32  * 
 33  * @event 
 34  * */
 35 Main.onLoad = function(){
 36 	document.getElementById("anchor").focus();
 37 	System.ready();
 38 	
 39 	Main.Runner = setTimeout(function(){
 40 		try{
 41 			System.Log.setMode(System.Log.MODE_DEVELOPMENT);
 42 			System.Log.setLevel(System.Log.LEVEL_DEBUG);
 43 			
 44 			Samsung.SEF.loadAll();
 45 			
 46 			System.Log.info("[main.js] Main.Runner Execution Complete!");
 47 		}
 48 		catch (error){
 49 			System.Log.error("[main.js] Main.Runner Execution Exception: " + error.message);
 50 		}
 51 		
 52 		clearTimeout(Main.Runner);
 53 		Main.Runner = null;
 54 	}, Main.PREPARE_DELAY);
 55 };
