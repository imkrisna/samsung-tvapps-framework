1 var System 	= System || new Object();
  2 System.Log 	= new Object();
  3 
  4 System.Log.MODE_DEBUGGING	= "SYSTEM_LOG_MODE_DEBUGGING";
  5 System.Log.MODE_DEVELOPMENT	= "SYSTEM_LOG_MODE_DEVELOPMENT";
  6 System.Log.MODE_PRODUCTION	= "SYSTEM_LOG_MODE_PRODUCTION";
  7 
  8 System.Log.LEVEL_VERBOSE	= 10;
  9 System.Log.LEVEL_DEBUG		= 20;
 10 System.Log.LEVEL_INFO		= 30;
 11 System.Log.LEVEL_WARNING	= 40;
 12 System.Log.LEVEL_ERROR		= 50;
 13 
 14 System.Log.COLOR_DEFAULT	= "#FFFFFF";
 15 System.Log.COLOR_DEBUG		= "#00FFFF";
 16 System.Log.COLOR_INFO		= "#3300FF";
 17 System.Log.COLOR_WARNING	= "#FFCC00";
 18 System.Log.COLOR_ERROR		= "#FF0000";
 19 
 20 System.Log.HTML_ID			= "framework-debug";
 21 System.Log.ALERT_PREFIX		= "### LOG ### ";
 22 System.Log.BUFFER_SIZE		= 50;
 23 
 24 System.Log.Buffer			= new Array();
 25 System.Log.BufferString		= STRING.EMPTY;
 26 System.Log.ActiveMode		= System.Log.MODE_DEVELOPMENT;
 27 System.Log.ActiveLevel		= System.Log.LEVEL_WARNING;
 28 
 29 
 30 System.Log.setMode = function(mode){
 31 	var debugHtml = System.Cache.getHtml(System.Log.HTML_ID);
 32 	
 33 	switch (mode){
 34 		case System.Log.MODE_DEBUGGING:
 35 			debugHtml.style.display = STRING.CSS_DISPLAY_BLOCK;
 36 			break;
 37 		default:
 38 			debugHtml.style.display = STRING.CSS_DISPLAY_NONE;
 39 			break;
 40 	}
 41 };
 42 
 43 System.Log.setLevel = function(level){
 44 	System.Log.ActiveLevel = level;
 45 };
 46 
 47 System.Log.verbose = function(message){
 48 	System.Log.log(message, System.Log.LEVEL_VERBOSE, System.Log.COLOR_DEFAULT);
 49 };
 50 
 51 System.Log.debug = function(message){
 52 	System.Log.log(message, System.Log.LEVEL_DEBUG, System.Log.COLOR_DEBUG);
 53 };
 54 
 55 System.Log.info = function(message){
 56 	System.Log.log(message, System.Log.LEVEL_INFO, System.Log.COLOR_INFO);
 57 };
 58 
 59 System.Log.warning = function(message){
 60 	System.Log.log(message, System.Log.LEVEL_WARNING, System.Log.COLOR_WARNING);
 61 };
 62 
 63 System.Log.error = function(message){
 64 	System.Log.log(message, System.Log.LEVEL_ERROR, System.Log.COLOR_ERROR);
 65 };
 66 
 67 System.Log.log = function(message, level, color){
 68 	if (!level)	level	= System.Log.LEVEL_VERBOSE;
 69 	if (!color) color 	= System.Log.COLOR_DEFAULT;
 70 	
 71 	if (System.Log.ActiveMode !== System.Log.MODE_PRODUCTION && level >= System.Log.ActiveLevel){		
 72 		alert(System.Log.ALERT_PREFIX + message);		
 73 		System.Log.Buffer.push("<span style=\"color:" + color + "\">" + message + "</span>");
 74 		
 75 		while (System.Log.Buffer.length > System.Log.BUFFER_SIZE){
 76 			System.Log.Buffer.splice(0, 1);
 77 		}
 78 		
 79 		if (System.Log.ActiveMode === System.Log.MODE_DEBUGGING){
 80 			System.Log.BufferString	= System.Log.Buffer.join("<br />");			
 81 			System.putInnerHtml(System.Log.HTML_ID, System.Log.BufferString);	
 82 		}				
 83 	}	
 84 };
